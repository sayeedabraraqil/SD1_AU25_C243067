import java.util.Scanner;
public class printtoN {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int i=0;
        for(i=1;i<=num;i++)
        {
            System.out.println(i);
        }
    sc.close();
}
    
}
