import java.util.Scanner;
public class multifplication {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int i=0;
        for(i=1;i<=10;i++)
        {
            System.out.println(num+" * "+i+" = "+(num*i));
        }
    sc.close();
}
    
}
