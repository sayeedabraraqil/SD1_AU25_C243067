import java.util.Scanner;
public class sum1toN {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int i=0,sum=0;
        for(i=0;i<=num;i++)
        {
            sum+=i;
        }
        System.out.println(sum);
        
    sc.close();
}
    
}
